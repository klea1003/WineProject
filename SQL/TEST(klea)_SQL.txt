create sequence seq_wine;

DROP TABLE TBL_WINE_TEST;

CREATE TABLE TBL_WINE_TEST (
    WNO NUMBER(20) NOT NULL,
    TITLE VARCHAR2(50),
    WINENERY VARCHAR2(50),
    GRAPES VARCHAR2(100),
    REGIONS VARCHAR2(50),
    ALLERGENS VARCHAR2(50),
    WINETYPE VARCHAR2(50),
    REGION VARCHAR2(50),
    COUNTRY VARCHAR2(50),
    PRICE NUMBER(20),
    ALCOHOL FLOAT,
    WINESTYLE VARCHAR2(50)
);

COMMIT;
